#! /bin/bash
function sayhello()
{
  echo "Hello,World"
  return 1
}

# 调用
. hello.sh
sayhello
result=$?
echo $result

# 因为bash的函数默认只能返回数值，且数值必须位于0->255，所以如果需要输入字符串的结果，就要使用折中的方式
# 比如下面函数是输出现在的时间，会echo输入字符串，取得变量的地方使用$()
function now()
{
	str=$(date "+%Y-%m-%d %H:%M:%S")
	echo $str
}
# 调用
n=$(now)

# 使用参数
function show(){
    echo "Tutorial: $1"
    echo "URL: $2"
    echo "Author: "$3
    echo "Total $# parameters"
}
#调用函数
show C# http://www.baidu.com/ Tom

- echo $SHELL 查看SHELL类型

- date 查看当前日期
- free -h 查看当前内存使用
- top 监控CPU，内存详细使用情况

Shell下的命令基本分为 : 内置命令(built-in function), 可执行文件(executable file), 别名(alias)， 使用which 可以确认命令对应哪个可执行文件

- which date , which 用于查看 date 这命令属于哪里，如 /bin/date
- alias freak="free -h" 设置别名
- type date, type 命令可以查看命令类型, type pwd, type freak
