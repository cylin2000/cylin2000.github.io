# 实现思路，使用脚本备份文件，如果发现差异，就保留备份，并通过ftp传到目标机器，使用Schedule Task定时轮循调用

#backup.bat
powershell .\backup.ps1

#backup.ps1
$scriptPath = Split-Path -Parent $MyInvocation.MyCommand.Definition
$timestamp = (Get-Date).ToString("yyyyMMddHHmmss")
sqlcmd -U sa -P xxxxx -S "(local)" -i "$scriptPath\backup.sql"
$preSize = Get-Content "$scriptPath\backup.size.txt"
$curSize = (Get-ChildItem .\express.bak).Length
if($preSize -ne $curSize){
    Set-Content "$scriptPath\backup.size.txt" $curSize
    ncftpput -u ftpuser -p ftppassword -m -R ftp.server.address.com "/DataBase/bak/express.$timestamp" express.bak
    Rename-Item "$scriptPath\express.bak" "$scriptPath\express.$timestamp.bak"
}
else{
    Write-Host "Same Size, remove the express.bak"
    Remove-Item "$scriptPath\express.bak"
}


# cleanup.ps1
$limit = (Get-Date).AddDays(-5)
$path = $scriptPath
# Delete files older than the $limit.
Get-ChildItem -Path $path -Recurse -Force -Include *.bak | Where-Object { !$_.PSIsContainer -and $_.CreationTime -lt $limit } | Remove-Item -Force
# Delete any empty directories left behind after deleting the old files.
Get-ChildItem -Path $path -Recurse -Force -Include *.bak | Where-Object { $_.PSIsContainer -and (Get-ChildItem -Path $_.FullName -Recurse -Force | Where-Object { !$_.PSIsContainer }) -eq $null } | Remove-Item -Force -Recurse

SELECT TOP 1000 
--创建时间 
QS.creation_time, 
--查询语句 
SUBSTRING(ST.text,(QS.statement_start_offset/2)+1, 
((CASE QS.statement_end_offset WHEN -1 THEN DATALENGTH(st.text) 
ELSE QS.statement_end_offset END - QS.statement_start_offset)/2) + 1 
) AS statement_text, 
--执行文本 
ST.text, 
--执行计划 
QS.total_worker_time, 
QS.last_worker_time, 
QS.max_worker_time, 
QS.min_worker_time 
FROM 
sys.dm_exec_query_stats QS 
--关键字 
CROSS APPLY 
sys.dm_exec_sql_text(QS.sql_handle) ST 
ORDER BY 
QS.creation_time DESC
